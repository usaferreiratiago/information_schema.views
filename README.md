# information_schema.views

SELECT 
    TABLE_CATALOG,
 CONCAT('SELECT * FROM',' ',TABLE_SCHEMA,'.',
lower(TABLE_NAME),';') AS 'INFORMATION SCHEMA',  
    VIEW_DEFINITION,
    CHECK_OPTION,
    IS_UPDATABLE,
    DEFINER,
    SECURITY_TYPE,
    CHARACTER_SET_CLIENT,
    COLLATION_CONNECTION
FROM
    information_schema.views;
